#include <iostream>

using namespace std;

// Função que retorna verdadeiro se o número for ímpar
bool ehImpar(int n) {
    return n % 2 != 0;
}

int main() {
    int numero;
    int soma = 0;

    cout << "Digite um numero inteiro: ";
    cin >> numero;

    for (int i = 1; i <= numero; i++) {
        if (ehImpar(i)) {
            soma += i;
        }
    }

    cout << "Soma dos numeros impares ate " << numero
         << " = " << soma << endl;

    return 0;
}

#include <iostream>
#include <string>
#include <cctype>

using namespace std;

// Função para verificar se é vogal
bool ehVogal(char c) {
    c = tolower(c);
    return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
}

// Função para contar vogais
int contarVogais(string nome) {
    int cont = 0;

    for (char c : nome) {
        if (isalpha(c) && ehVogal(c)) {
            cont++;
        }
    }

    return cont;
}

// Função para contar consoantes
int contarConsoantes(string nome) {
    int cont = 0;

    for (char c : nome) {
        if (isalpha(c) && !ehVogal(c)) {
            cont++;
        }
    }

    return cont;
}

int main() {
    string nome;

    cout << "Digite o nome completo: ";
    getline(cin, nome);

    cout << "Quantidade de vogais: " << contarVogais(nome) << endl;
    cout << "Quantidade de consoantes: " << contarConsoantes(nome) << endl;

    return 0;
}

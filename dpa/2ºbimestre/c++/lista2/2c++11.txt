#include <iostream>

using namespace std;

int main() {
    int vetor[10];

    cout << "Digite 10 numeros:" << endl;

    // Leitura do vetor
    for (int i = 0; i < 10; i++) {
        cin >> vetor[i];
    }

    // Substituição dos negativos por zero
    for (int i = 0; i < 10; i++) {
        if (vetor[i] < 0) {
            vetor[i] = 0;
        }
    }

    // Exibição do vetor
    cout << "Vetor final:" << endl;
    for (int i = 0; i < 10; i++) {
        cout << vetor[i] << " ";
    }

    cout << endl;

    return 0;
}

#include <iostream>

using namespace std;

int main() {
    int vetor[15];
    int pares = 0;
    int impares = 0;

    cout << "Digite 15 numeros:" << endl;

    // Leitura do vetor
    for (int i = 0; i < 15; i++) {
        cin >> vetor[i];
    }

    // Contagem de pares e ímpares
    for (int i = 0; i < 15; i++) {
        if (vetor[i] % 2 == 0) {
            pares++;
        } else {
            impares++;
        }
    }

    cout << "Quantidade de pares: " << pares << endl;
    cout << "Quantidade de impares: " << impares << endl;

    return 0;
}

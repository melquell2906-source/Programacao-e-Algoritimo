#include <iostream>

using namespace std;

int main() {
    char vetor[20];
    char letra;
    int cont = 0;

    cout << "Digite 20 letras:" << endl;

    // Leitura do vetor
    for (int i = 0; i < 20; i++) {
        cin >> vetor[i];
    }

    cout << "Digite uma letra para buscar: ";
    cin >> letra;

    // Contagem de ocorrências
    for (int i = 0; i < 20; i++) {
        if (vetor[i] == letra) {
            cont++;
        }
    }

    cout << "A letra '" << letra << "' aparece "
         << cont << " vezes." << endl;

    return 0;
}

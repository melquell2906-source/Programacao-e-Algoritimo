#include <iostream>

using namespace std;

int main() {
    int vetor[20];
    int indice = -1;

    cout << "Digite 20 numeros:" << endl;

    // Leitura do vetor
    for (int i = 0; i < 20; i++) {
        cin >> vetor[i];
    }

    // Busca do primeiro negativo
    for (int i = 0; i < 20; i++) {
        if (vetor[i] < 0) {
            indice = i;
            break; // para no primeiro encontrado
        }
    }

    if (indice != -1) {
        cout << "Primeiro numero negativo esta no indice: " << indice << endl;
    } else {
        cout << "Nao ha numeros negativos no vetor." << endl;
    }

    return 0;
}

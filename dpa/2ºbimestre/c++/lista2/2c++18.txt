#include <iostream>
#include <cctype>

using namespace std;

int main() {
    char letras[20];
    int vogais[5] = {0, 0, 0, 0, 0}; // a, e, i, o, u

    cout << "Digite 20 letras:" << endl;

    // Leitura das letras
    for (int i = 0; i < 20; i++) {
        cin >> letras[i];
    }

    // Contagem das vogais
    for (int i = 0; i < 20; i++) {
        char c = tolower(letras[i]);

        switch (c) {
            case 'a': vogais[0]++; break;
            case 'e': vogais[1]++; break;
            case 'i': vogais[2]++; break;
            case 'o': vogais[3]++; break;
            case 'u': vogais[4]++; break;
        }
    }

    // Exibição
    cout << "Quantidade de vogais:" << endl;
    cout << "A: " << vogais[0] << endl;
    cout << "E: " << vogais[1] << endl;
    cout << "I: " << vogais[2] << endl;
    cout << "O: " << vogais[3] << endl;
    cout << "U: " << vogais[4] << endl;

    return 0;
}

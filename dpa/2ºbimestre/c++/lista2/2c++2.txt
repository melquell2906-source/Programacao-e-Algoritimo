#include <iostream>

using namespace std;

// Função para calcular potência
int potencia(int base, int expoente) {
    int resultado = 1;

    for (int i = 1; i <= expoente; i++) {
        resultado *= base;
    }

    return resultado;
}

int main() {
    int base, expoente;

    cout << "Digite a base: ";
    cin >> base;

    cout << "Digite o expoente: ";
    cin >> expoente;

    cout << base << "^" << expoente << " = "
         << potencia(base, expoente) << endl;

    return 0;
}
